скачать с https://gitforwindows.org/

установить

запустить консоль bash git

создать папку для репозитория

cd диск:/папка(репозитория)      в консоле bash git

gitinit                          в консоле bash git


git status                    посмотреть состояние файлов в репозитории

git add .                      добавить все файлы под контроль



git config --global user.name "никнейм"
git config --global user.email "почта"

git commit -m "комментарий"
